3.1 4.1 Prueba de Kruskal-Wallis

# Kruskal-Wallis: Tiempo en hospital vs Readmision
kruskal_time <- kruskal.test(time_in_hospital ~ readmitted, 
                            data = diabetic_data %>% filter(!is.na(readmitted)))
cat("Prueba de Kruskal-Wallis: Tiempo en Hospital vs Readmision\n")
## Prueba de Kruskal-Wallis: Tiempo en Hospital vs Readmision
print(kruskal_time)
## 
##  Kruskal-Wallis rank sum test
## 
## data:  time_in_hospital by readmitted
## Kruskal-Wallis chi-squared = 469.24, df = 2, p-value < 2.2e-16

3.4 4.4 Pruebas T para Muestras Independientes

# Crear variable binaria para readmision temprana (<30 dias)
diabetic_data <- diabetic_data %>%
  mutate(readmission_30 = ifelse(readmitted == "<30", 1, 0))

# Prueba T: Comparar numero de medicamentos entre readmitidos temprano vs otros
t_test_meds <- t.test(num_medications ~ readmission_30, 
                      data = diabetic_data, na.action = na.omit)
cat("Prueba T: Numero de Medicamentos - Readmision <30 dias vs Otros\n")
## Prueba T: Numero de Medicamentos - Readmision <30 dias vs Otros
print(t_test_meds)
